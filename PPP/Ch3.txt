#include "std_lib_facilities.h"

int main()
{
	float Num1 = 0.0, Num2 = 0.0;
	string Operand = "";
	cout << "Please, enter operand and two numbers: ";
	cin >> Operand >> Num1 >> Num2;

	if(Operand == "+" || Operand == "plus") cout << Num1 + Num2;
	if (Operand == "-" || Operand == "minus") cout << Num1 - Num2;
	if (Operand == "*" || Operand == "mult") cout << Num1 * Num2;
	if (Operand == "/" || Operand == "div")
	{
		if (Num2 != 0) cout << Num1 / Num2;
		else cout << "Impossible to divide by zero!\n";
	}
	return 0;
}
